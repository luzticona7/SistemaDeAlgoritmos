Merge Sort: Es un algoritmo de ordenamiento eficiente que utiliza el enfoque "divide y vencerás". Divide el array en mitades, ordena cada mitad recursivamente y luego las combina.

// Función principal de Merge Sort
function mergeSort(arr, left, right) {
  if (left < right) {
    const mid = Math.floor((left + right) / 2);
    mergeSort(arr, left, mid);
    mergeSort(arr, mid + 1, right);
    merge(arr, left, mid, right);
  }
}
